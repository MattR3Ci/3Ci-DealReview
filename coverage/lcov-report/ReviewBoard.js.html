        initEditors: function () {
            const self = this;
            const toolbarOptions = [
                ['bold', 'italic', 'underline'],
                [{ 'list': 'ordered' }, { 'list': 'bullet' }],
                ['clean']

            this.quillNotes = new Quill('#editor-notes', { theme: 'snow', modules: { toolbar: toolbarOptions }, placeholder: 'Enter feedback notes...' });
            this.quillAction = new Quill('#editor-action', { theme: 'snow', modules: { toolbar: toolbarOptions }, placeholder: 'List required actions for resubmission...' });
 
            this.quillNotes.on('text-change', function () { self.state.feedback.notes = self.quillNotes.root.innerHTML; });
 
            this.quillAction.on('text-change', function () {
                self.state.feedback.actionSummary = self.quillAction.root.innerHTML;
 
                const rawText = self.quillAction.getText().trim();
                const isRejecting = self.state.feedback.decision === 'reject';
                const isActionMissing = isRejecting && rawText.length === 0;
 
                if (isActionMissing) {
                    $('#editor-action-container').css({ 'border-color': '#EF4444', 'border-width': '2px' });
                    $('#action-summary-warning').show();
                } else {
                    $('#editor-action-container').css({ 'border-color': 'var(--border-default, #E2E8F0)', 'border-width': '1px' });
                    $('#action-summary-warning').hide();
                }
 
                const isSubmitDisabled = !self.state.feedback.decision || isActionMissing;
                $('#btn-submit-review').prop('disabled', isSubmitDisabled).css('opacity', isSubmitDisabled ? '0.5' : '1');
            });

        /* =========================================================

        bindEvents: function () {
            const self = this;
 
            $(document).off('click', '#btn-back-dashboard').on('click', '#btn-back-dashboard', function () { window.location.href = '/'; });
            $(document).off('click', '.dr-tab-pill').on('click', '.dr-tab-pill', function () { self.selectDeal($(this).data('id')); });
 
            $(document).off('click', '.btn-decision').on('click', '.btn-decision', function () {
                self.state.feedback.decision = $(this).data('vote');
                self.renderDealDetails();
            });
 
            $(document).off('click', '#btn-exc-approve').on('click', '#btn-exc-approve', function () {
                self.state.feedback.exceptionApproved = true;
                self.renderDealDetails();
            });
            $(document).off('click', '#btn-exc-reject').on('click', '#btn-exc-reject', function () {
                self.state.feedback.exceptionApproved = false;
                self.renderDealDetails();
            });
 
            $(document).off('click', '#btn-submit-review').on('click', '#btn-submit-review', async function () {
                const decision = self.state.feedback.decision;
                if (!decision) return;
 
                const F = window.SCHEMA.FIELDS.DEALS;
                const C = window.SCHEMA.CHOICES.STATUS;
                const deal = self.state.deals.find(d => d[F.DEALID] === self.state.selectedDealId);
 
                if (decision === 'reject' && !self.state.feedback.actionSummary.trim()) {
                    alert("Validation Error: An 'Action Required Summary' is mandatory when rejecting.");
                    return;
                }
 
                const $btn = $(this);
                $btn.html('<i class="fa fa-spinner fa-spin"></i> Recording...').prop('disabled', true);
 
                try {
                    const updateData = {
                        [F.STATUS]: decision === 'approve' ? C.APPROVED : C.REJECTED,
                        [F.REVIEW_NOTES]: self.state.feedback.notes || "",
                        [F.ACTION_REQUIRED]: self.state.feedback.actionSummary || ""
                    };
                    if (deal[F.EXCEPTION_REQ]) updateData[F.EXCEPTION_APP] = self.state.feedback.exceptionApproved;
 
                    const auditPayload = {
                        dealId: String(deal[F.EXTERNALKEY]),
                        decision: decision === 'approve' ? "Approve" : "Reject",
                        notes: self.state.feedback.notes,
                        actionSummary: self.state.feedback.actionSummary,
                        reviewer: window.USER_CONTEXT?.fullname || "Review Board"
                    };
 
                    const [dvSuccess] = await Promise.all([
                        window.SubmissionService.patchDeal(self.state.selectedDealId, updateData),
                        fetch(window.SCHEMA.ENDPOINTS.RECORD_REVIEW_DECISION, {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json' },
                            body: JSON.stringify(auditPayload)
                        })
                    ]);
 
                    if (dvSuccess) {
                        const completedDealId = self.state.selectedDealId;
                        self.state.deals = self.state.deals.filter(d => d[F.DEALID] !== self.state.selectedDealId);
                        $('#ui-review-count').text(`TODAY'S REVIEWS (${self.state.deals.length})`);
 
                        if (self.state.deals.length > 0) {
                            await self.selectDeal(self.state.deals[0][F.DEALID]);
                        } else {
                            self.state.selectedDealId = null;
                            self.renderEmptyState();
                            $('#deal-tabs-container').empty();
                        }
                        
                        window.ImpactTracker.showModal(completedDealId, decision);
                    }
                } catch (e) {
                    console.error("Submission Error:", e);
                    alert("Failed to record decision. Please try again.");
                    $btn.html('Record Decision').prop('disabled', false);
                }
            });
        }
    };
 
    window.ReviewBoardApp = ReviewBoardApp;

        const checkDependencies = setInterval(() => {
            if (window.SCHEMA && window.UIComponents && window.SubmissionCalc && window.SubmissionService) {
                clearInterval(checkDependencies);
                ReviewBoardApp.init();
            }
        }, 50);
    });
